import pandas as pd
# Below is for an interactive display of DataFrame in Colab
from google.colab import data_table
data_table.enable_dataframe_formatter()

emp = pd.read_csv("https://bcdanl.github.io/data/employment.csv")
emp.info()
mean_sal=emp['Salary'].mean()
emp['Salary']=emp['Salary'].fillna(mean_sal)

emp= emp.astype({
    "Gender":'category',
    "Start Date":"datetime64",
    "Mgmt":"bool",
    "Team":"category"
})
emp.info()
#finding non-missing values
# Below two are equivalent.
emp["Team"].notna()
~emp["Team"].isna()
#how many missing values?
emp['Team'].isna().sum
emp["Mgmt"].isna().sum()
emp["Mgmt"].value_counts()
emp["Mgmt"].value_counts(dropna = False)
# .dropna() removes observations that are NaN or NaT - default is with how='any'
emp.dropna()
# .dropna(how='all') removes observations where all values are missing
emp.dropna(how = "all")
# .dropna(subset=[]) targets missing values in certain variable(s)
emp.dropna(subset = ["Gender"])
emp.dropna(subset = ["Start Date", "Salary"]).info()
# ^ you can use multiple .functions at once
# .dropna(thresh= ) gives specified min threshold of nonmissing values that an
#observation has to have in order for the variable to not get removed
emp.dropna(thresh = 4)
#DEALING WITH DUPLICATED VALUES
emp["Team"].duplicated()
#returns bool series that identifies duplicates
#the first occurrence of a duplicated value will be false bc there hasn't been
#a duplicate yet
emp["Team"].duplicated(keep = "first")
#default
emp["Team"].duplicated(keep = "last")
#keeping last occurrence of duplicated values
~emp["Team"].duplicated()
#negates duplicated... returns false instead of true for duplicates
#can be useful to keep only unique values
# .drop_duplicates() removes observations in which ALL values are the same as
#another
emp.drop_duplicates()
#there are no duplicates in emp
# Sample DataFrame with duplicate observations
data = {
    'Name': ['John', 'Anna', 'John', 'Mike', 'Anna'],
    'Age': [28, 23, 28, 32, 23],
    'City': ['New York', 'Paris', 'New York', 'London', 'Paris']
}

# pd.DataFrame( Series, List, or Dict ) creates a DataFrame
df=pd.DataFrame(data)
df
df.drop_duplicates()
# .drop_duplicates(subset=[]) finds the first occurrence of a value in a certain
#variable and keeps only those observations
emp.drop_duplicates(subset=['Team'])
emp.drop_duplicates(subset = ["Gender", "Team"])
#shows 3 observations per team based on gender (male, female, missing)
emp.drop_duplicates(subset = ["Team"], keep = "last")
emp.drop_duplicates(subset = ["Team"], keep = False) #keep=False gets rid of
#anything in Team that is duplicated
#drops all observations bc all team values in team are duplicated
emp.drop_duplicates(['First Name'])
emp.query("`First Name`=='David'")
emp.query("`First Name`=='Douglas' & `Gender`=='Male'")
